```typescript
/**
 * Test /api/evaluate-interview endpoint
 */

import { sampleInterview, expectedOutput } from './test-data';

async function testEvaluateInterviewAPI() {
  console.log('\n=== TEST: /api/evaluate-interview endpoint ===\n');
  
  const startTime = Date.now();
  
  try {
    // Make API call
    const response = await fetch('http://localhost:3000/api/evaluate-interview', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(sampleInterview)
    });
    
    const duration = Date.now() - startTime;
    
    // Check response status
    if (!response.ok) {
      const error = await response.json();
      console.error('❌ API call failed:', error);
      return false;
    }
    
    const result = await response.json();
    
    // Validate response structure
    console.log('✓ API call succeeded');
    console.log(`✓ Duration: ${duration}ms`);
    
    if (!result.success) {
      console.error('❌ Response success is false');
      return false;
    }
    console.log('✓ Response success: true');
    
    if (!result.evaluationId) {
      console.error('❌ Missing evaluationId');
      return false;
    }
    console.log(`✓ Evaluation ID: ${result.evaluationId}`);
    
    if (!result.preview || typeof result.preview.overallScore !== 'number') {
      console.error('❌ Missing or invalid preview.overallScore');
      return false;
    }
    console.log(`✓ Overall Score: ${result.preview.overallScore}/5.0`);
    
    if (result.preview.questionCount !== expectedOutput.questionCount) {
      console.error(`❌ Expected ${expectedOutput.questionCount} questions, got ${result.preview.questionCount}`);
      return false;
    }
    console.log(`✓ Question count: ${result.preview.questionCount}`);
    
    // Check Firestore document structure
    console.log('\n--- Fetching Firestore document ---\n');
    
    const evalResponse = await fetch(`http://localhost:3000/api/evaluation/${result.evaluationId}`);
    if (evalResponse.ok) {
      const evaluation = await evalResponse.json();
      
      // Validate structure
      if (!evaluation.centralRubricScores) {
        console.error('❌ Missing centralRubricScores');
        return false;
      }
      console.log('✓ Has centralRubricScores');
      
      // Check all dimensions present
      for (const dim of expectedOutput.centralRubricDimensions) {
        if (!(dim in evaluation.centralRubricScores)) {
          console.error(`❌ Missing dimension: ${dim}`);
          return false;
        }
        const score = evaluation.centralRubricScores[dim];
        if (score < 1 || score > 5) {
          console.error(`❌ Invalid score for ${dim}: ${score} (should be 1-5)`);
          return false;
        }
      }
      console.log('✓ All 6 central rubric dimensions present (1-5 scale)');
      
      if (!Array.isArray(evaluation.strengths) || evaluation.strengths.length !== 3) {
        console.error(`❌ Expected 3 strengths, got ${evaluation.strengths?.length || 0}`);
        return false;
      }
      console.log(`✓ Has 3 strengths`);
      
      if (!Array.isArray(evaluation.improvements) || evaluation.improvements.length !== 3) {
        console.error(`❌ Expected 3 improvements, got ${evaluation.improvements?.length || 0}`);
        return false;
      }
      console.log(`✓ Has 3 improvements`);
      
      if (!Array.isArray(evaluation.questions) || evaluation.questions.length !== 3) {
        console.error(`❌ Expected 3 questions, got ${evaluation.questions?.length || 0}`);
        return false;
      }
      console.log(`✓ Has 3 questions with evaluations`);
      
      // Check each question has judge evaluations
      for (let i = 0; i < evaluation.questions.length; i++) {
        const q = evaluation.questions[i];
        if (!Array.isArray(q.judgeEvaluations) || q.judgeEvaluations.length !== 2) {
          console.error(`❌ Question ${i + 1} should have 2 judge evaluations, got ${q.judgeEvaluations?.length || 0}`);
          return false;
        }
      }
      console.log('✓ Each question has 2 judge evaluations');
    }
    
    console.log('\n✅ ALL TESTS PASSED\n');
    return true;
    
  } catch (error) {
    console.error('❌ Test failed with error:', error);
    return false;
  }
}

// Run test
testEvaluateInterviewAPI().then(success => {
  process.exit(success ? 0 : 1);
});
```

```typescript
/**
 * End-to-end integration test
 */

import StarJudge from '@/lib/evaluation/judges/StarJudge';
import CompetencyJudge from '@/lib/evaluation/judges/CompetencyJudge';
import { Aggregator } from '@/lib/evaluation/Aggregator';
import { sampleInterview } from './test-data';

async function integrationTest() {
  console.log('\n=== INTEGRATION TEST: Full Evaluation Flow ===\n');
  
  const apiKey = process.env.GOOGLE_GENERATIVE_AI_API_KEY;
  if (!apiKey) {
    console.error('❌ GOOGLE_GENERATIVE_AI_API_KEY not set');
    return false;
  }
  
  try {
    const startTime = Date.now();
    
    // Step 1: Prepare questions
    const questionInputs = sampleInterview.questions.map((q, idx) => ({
      questionId: `question-${idx}`,
      question: q,
      answer: "",
      context: sampleInterview.interviewContext
    }));
    
    console.log(`Testing with ${questionInputs.length} questions...\n`);
    
    // Step 2: StarJudge evaluation
    console.log('Step 1/3: StarJudge evaluation...');
    const starJudge = new StarJudge(apiKey);
    const starStart = Date.now();
    const starEvaluations = await starJudge.evaluateMultiple(questionInputs);
    console.log(`✓ StarJudge complete (${Date.now() - starStart}ms)`);
    console.log(`  - Evaluated ${starEvaluations.length} questions`);
    
    // Step 3: CompetencyJudge evaluation
    console.log('\nStep 2/3: CompetencyJudge evaluation...');
    const competencyJudge = new CompetencyJudge(apiKey);
    const compStart = Date.now();
    const competencyEvaluations = await competencyJudge.evaluateMultiple(questionInputs);
    console.log(`✓ CompetencyJudge complete (${Date.now() - compStart}ms)`);
    console.log(`  - Evaluated ${competencyEvaluations.length} questions`);
    
    // Step 4: Combine evaluations
    const judgeEvaluationsPerQuestion = sampleInterview.questions.map((_, idx) => [
      starEvaluations[idx],
      competencyEvaluations[idx]
    ]);
    
    // Step 5: Aggregator
    console.log('\nStep 3/3: Aggregator processing...');
    const aggregator = new Aggregator();
    const aggStart = Date.now();
    const result = await aggregator.aggregateInterview(
      {
        fullTranscript: sampleInterview.fullTranscript,
        questions: sampleInterview.questions,
        interviewContext: sampleInterview.interviewContext,
        judgeEvaluationsPerQuestion
      },
      apiKey
    );
    console.log(`✓ Aggregator complete (${Date.now() - aggStart}ms)`);
    
    const totalTime = Date.now() - startTime;
    
    // Validate results
    console.log('\n--- Validation ---\n');
    
    console.log(`Overall Score: ${result.overallScore}/5.0`);
    console.log(`Confidence: ${(result.confidenceScore * 100).toFixed(0)}%`);
    console.log(`\nCentral Rubric Scores:`);
    console.log(`  - Coherence: ${result.centralRubricScores.coherenceStructure}/5`);
    console.log(`  - Fluency: ${result.centralRubricScores.fluency}/5`);
    console.log(`  - Lexical: ${result.centralRubricScores.lexical}/5`);
    console.log(`  - Technical: ${result.centralRubricScores.technical}/5`);
    console.log(`  - Soft Skill: ${result.centralRubricScores.softSkill}/5`);
    console.log(`  - Cultural: ${result.centralRubricScores.cultural}/5`);
    
    console.log(`\nStrengths (${result.strengths.length}):`);
    result.strengths.forEach((s, i) => console.log(`  ${i + 1}. ${s}`));
    
    console.log(`\nImprovements (${result.improvements.length}):`);
    result.improvements.forEach((imp, i) => console.log(`  ${i + 1}. ${imp}`));
    
    console.log(`\n--- Performance ---\n`);
    console.log(`Total processing time: ${totalTime}ms`);
    console.log(`Average per question: ${Math.round(totalTime / questionInputs.length)}ms`);
    
    // Performance check
    if (totalTime > 60000) {
      console.warn(`⚠️  Processing took longer than expected: ${totalTime}ms`);
    } else {
      console.log(`✓ Processing time acceptable`);
    }
    
    console.log('\n✅ INTEGRATION TEST PASSED\n');
    return true;
    
  } catch (error) {
    console.error('❌ Integration test failed:', error);
    return false;
  }
}

// Run test
integrationTest().then(success => {
  process.exit(success ? 0 : 1);
});
```

## TROUBLESHOOTING

### Test fails with "API key not configured":
```bash
# Set environment variable
export GOOGLE_GENERATIVE_AI_API_KEY="your-actual-key"

# Or add to .env.local
echo "GOOGLE_GENERATIVE_AI_API_KEY=your-key" >> .env.local
```

### Test fails with "Cannot find module":
```bash
# Install tsx if not present
npm install -D tsx

# Or use ts-node
npm install -D ts-node
npx ts-node tests/evaluation/api.test.ts
```

### Test times out:
```bash
# Increase timeout (in test file):
// Add at top of test
jest.setTimeout(60000); // 60 seconds

# Or use --timeout flag
npx tsx --timeout=60000 tests/evaluation/api.test.ts
```

### Firestore permission error:
```bash
# Check Firebase Admin credentials
echo $FIREBASE_PRIVATE_KEY
echo $FIREBASE_CLIENT_EMAIL

# Make sure they're set in .env.local
```
