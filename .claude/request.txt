```tsx
'use client';

import React, { useState } from 'react';
import { ChevronDown, ChevronUp } from 'lucide-react'; // Add these imports
import ScoreIndicator from './ScoreIndicator';

interface JudgeCardProps {
  evaluation: JudgeEvaluation;
  defaultExpanded?: boolean;
}

export default function JudgeCard({ 
  evaluation, 
  defaultExpanded = false // CHANGE: Default to collapsed
}: JudgeCardProps) {
  // Toggle state - starts COLLAPSED
  const [isExpanded, setIsExpanded] = useState(defaultExpanded);
  
  return (
    <div className="border-2 border-gray-700 rounded-xl bg-gray-900/80 overflow-hidden">
      {/* HEADER - Always visible, clickable to toggle */}
      <button
        onClick={() => setIsExpanded(!isExpanded)}
        className="w-full px-6 py-4 flex justify-between items-center hover:bg-gray-800/50 transition-colors group"
      >
        <div className="flex items-center gap-4">
          {/* Judge Label Badge */}
          <div className="px-3 py-1 rounded-full bg-gradient-to-r from-blue-600 to-indigo-600 text-white text-sm font-semibold">
            {evaluation.judgeName}
          </div>
          
          {/* Judge ID (subtle) */}
          <span className="text-xs text-gray-500">
            {evaluation.judgeId}
          </span>
        </div>
        
        <div className="flex items-center gap-4">
          {/* Overall Score - Large and prominent */}
          <div className="text-right">
            <div className="text-3xl font-bold text-white">
              {evaluation.overallScore.toFixed(2)}
              <span className="text-base text-gray-400 font-normal ml-1">/4.0</span>
            </div>
            <div className="text-xs text-gray-400">
              {(evaluation.confidence * 100).toFixed(0)}% confidence
            </div>
          </div>
          
          {/* Expand/Collapse Icon */}
          {isExpanded ? (
            <ChevronUp className="w-5 h-5 text-gray-400 group-hover:text-white transition-colors" />
          ) : (
            <ChevronDown className="w-5 h-5 text-gray-400 group-hover:text-white transition-colors" />
          )}
        </div>
      </button>
      
      {/* EXPANDABLE CONTENT - Hidden by default */}
      {isExpanded && (
        <div className="px-6 pb-6 space-y-6 border-t border-gray-700 pt-6 bg-gray-800/30">
          {/* Label to clarify this is detailed breakdown */}
          <div className="flex items-center gap-2 text-sm text-gray-400">
            <span className="font-semibold text-white">{evaluation.judgeName}</span>
            <span>·</span>
            <span>Detailed Breakdown</span>
          </div>
          
          {/* Dimension Scores */}
          <div className="space-y-4">
            <h4 className="font-semibold text-white text-sm uppercase tracking-wide">
              Dimension Scores
            </h4>
            {evaluation.dimensions.map((dim, idx) => (
              <div key={idx} className="space-y-2">
                <ScoreIndicator
                  score={dim.score}
                  maxScore={dim.maxScore}
                  label={dim.name}
                  size="md"
                />
                <div className="pl-4 space-y-2">
                  <p className="text-sm text-gray-300">
                    <span className="font-medium text-white">Reasoning:</span> {dim.reasoning}
                  </p>
                  {dim.evidence.length > 0 && (
                    <div className="text-sm">
                      <span className="font-medium text-white">Evidence:</span>
                      <ul className="mt-1 space-y-1">
                        {dim.evidence.map((quote, qIdx) => (
                          <li key={qIdx} className="text-gray-400 italic pl-4 border-l-2 border-gray-600">
                            "{quote}"
                          </li>
                        ))}
                      </ul>
                    </div>
                  )}
                </div>
              </div>
            ))}
          </div>
          
          {/* Strengths */}
          <div className="space-y-2">
            <h4 className="font-semibold text-white flex items-center gap-2 text-sm uppercase tracking-wide">
              <span className="w-2 h-2 rounded-full bg-green-500"></span>
              Strengths
            </h4>
            <ul className="space-y-2">
              {evaluation.strengths.map((strength, idx) => (
                <li key={idx} className="text-sm text-gray-300 pl-7 relative">
                  <span className="absolute left-0 text-green-500">✓</span>
                  {strength}
                </li>
              ))}
            </ul>
          </div>
          
          {/* Improvements */}
          <div className="space-y-2">
            <h4 className="font-semibold text-white flex items-center gap-2 text-sm uppercase tracking-wide">
              <span className="w-2 h-2 rounded-full bg-blue-500"></span>
              Areas for Improvement
            </h4>
            <ul className="space-y-2">
              {evaluation.improvements.map((improvement, idx) => (
                <li key={idx} className="text-sm text-gray-300 pl-7 relative">
                  <span className="absolute left-0 text-blue-500">→</span>
                  {improvement}
                </li>
              ))}
            </ul>
          </div>
        </div>
      )}
    </div>
  );
}
```
