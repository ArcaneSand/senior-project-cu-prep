```typescript
import { generateObject } from 'ai';
import { google } from '@ai-sdk/google';

/**
 * Aggregator - Combines judge evaluations with AI-powered central rubric scoring
 */
export class Aggregator {
  
  /**
   * NEW: Aggregate entire interview evaluation
   * 
   * @param params - Interview evaluation parameters
   * @param apiKey - Google AI API key for Gemini calls
   * @returns Complete aggregated evaluation
   */
  async aggregateInterview(
    params: {
      fullTranscript: string;
      questions: string[];
      interviewContext: InterviewContext;
      judgeEvaluationsPerQuestion: JudgeEvaluation[][];
    },
    apiKey: string
  ): Promise<AggregatedEvaluationResult> {
    console.log('Aggregator: Starting interview-level aggregation...');
    
    const model = google('gemini-2.0-flash-exp', { apiKey });
    
    // Step 1: Parse transcript to extract Q&A pairs
    const parsedQA = await this.parseTranscript(
      params.fullTranscript,
      params.questions,
      model
    );
    
    // Step 2: Combine with judge evaluations
    const questionEvaluations: QuestionEvaluation[] = parsedQA.map((qa, idx) => ({
      questionId: qa.questionId,
      question: qa.question,
      answer: qa.answer,
      judgeEvaluations: params.judgeEvaluationsPerQuestion[idx] || []
    }));
    
    // Step 3: Score against central rubric
    const centralScores = await this.scoreCentralRubric(
      questionEvaluations,
      params.interviewContext,
      model
    );
    
    // Step 4: Calculate overall score (mean of 6 dimensions)
    const overallScore = this.mean([
      centralScores.coherenceStructure,
      centralScores.fluency,
      centralScores.lexical,
      centralScores.technical,
      centralScores.softSkill,
      centralScores.cultural
    ]);
    
    // Step 5: Synthesize feedback
    const allJudgeEvaluations = questionEvaluations.flatMap(q => q.judgeEvaluations);
    const synthesized = await this.synthesizeWithAI(
      {
        centralScores,
        judgeEvaluations: allJudgeEvaluations,
        questionEvaluations
      },
      model
    );
    
    // Step 6: Calculate agreement
    const judgeAgreement = this.calculateJudgeAgreement(allJudgeEvaluations);
    
    // Step 7: Calculate confidence
    const confidenceScore = this.calculateConfidence(
      allJudgeEvaluations,
      centralScores,
      judgeAgreement.overall
    );
    
    console.log(`Aggregator: Complete. Overall Score: ${overallScore.toFixed(2)}/5.0`);
    
    return {
      questionEvaluations,
      centralRubricScores: centralScores,
      overallScore,
      strengths: synthesized.strengths,
      improvements: synthesized.improvements,
      judgeAgreement,
      confidenceScore
    };
  }
  
  /**
   * Parse transcript to extract Q&A pairs using AI
   */
  private async parseTranscript(
    fullTranscript: string,
    questions: string[],
    model: ReturnType<typeof google>
  ): Promise<Array<{ questionId: string; question: string; answer: string }>> {
    console.log('Aggregator: Parsing transcript with AI...');
    
    const schema = z.object({
      questionAnswers: z.array(z.object({
        questionId: z.string(),
        question: z.string(),
        answer: z.string()
      }))
    });
    
    const { object } = await generateObject({
      model,
      schema,
      prompt: `
You are parsing a mock interview transcript to extract question-answer pairs.

INTERVIEW QUESTIONS (in order):
${questions.map((q, i) => `${i + 1}. ${q}`).join('\n')}

FULL TRANSCRIPT:
${fullTranscript}

TASK:
Extract the candidate's answer to EACH question from the transcript.
The transcript includes interviewer (assistant/system) and candidate (user) messages.

For each question, combine ALL candidate responses related to that question into a single answer.

Return JSON:
{
  "questionAnswers": [
    {
      "questionId": "question-0",
      "question": "exact question text",
      "answer": "candidate's complete answer"
    },
    ...
  ]
}

IMPORTANT:
- questionId format: "question-0", "question-1", etc.
- Include the FULL question text exactly as provided
- Combine multiple candidate turns if answering the same question
- If candidate didn't answer, set answer to "" (empty string)
      `.trim()
    });
    
    return object.questionAnswers;
  }
  
  /**
   * Score interview against central rubric using AI
   */
  private async scoreCentralRubric(
    questionEvaluations: QuestionEvaluation[],
    context: InterviewContext,
    model: ReturnType<typeof google>
  ): Promise<CentralRubricScores> {
    console.log('Aggregator: Scoring against central rubric...');
    
    const allQA = questionEvaluations.map((q, i) => 
      `Q${i + 1}: ${q.question}\nA${i + 1}: ${q.answer}`
    ).join('\n\n');
    
    const rubricPrompt = formatRubricForPrompt();
    
    const schema = z.object({
      coherenceStructure: z.number().min(1).max(5),
      fluency: z.number().min(1).max(5),

   * Synthesize strengths and improvements using AI

  private async synthesizeWithAI(

      centralScores: CentralRubricScores;

      questionEvaluations: QuestionEvaluation[];
    },
    model: ReturnType<typeof google>

CENTRAL RUBRIC SCORES (1-5 scale):
- Coherence Structure: ${params.centralScores.coherenceStructure}/5
- Fluency: ${params.centralScores.fluency}/5
- Lexical: ${params.centralScores.lexical}/5
- Technical: ${params.centralScores.technical}/5
- Soft Skill: ${params.centralScores.softSkill}/5
- Cultural: ${params.centralScores.cultural}/5

JUDGE FEEDBACK (raw strengths):
${allStrengths.slice(0, 20).join('; ')}

JUDGE FEEDBACK (raw improvements):
${allImprovements.slice(0, 20).join('; ')}

TASK:
Extract the TOP 3 STRENGTHS and TOP 3 IMPROVEMENTS.

REQUIREMENTS:
- Prioritize themes mentioned by MULTIPLE judges
- Reference specific scores from central rubric
- Be specific and actionable (NOT generic)
- Include evidence when possible
- Format as complete sentences

GOOD EXAMPLES:
Strengths:
- "Excellent STAR structure in behavioral responses (Coherence: 4.2/5, mentioned by both judges)"
- "Strong technical depth with specific examples (Technical: 4.5/5)"
- "Minimal filler words and confident delivery (Fluency: 4.0/5)"

Improvements:
- "Could strengthen cultural fit examples (Cultural: 3.2/5, only 1/3 questions addressed)"
- "Some technical answers lacked quantified results (CompetencyJudge flagged Q2)"
- "Occasional self-correction in explanations (Fluency dimension, 2/3 questions)"

Return exactly 3 strengths and 3 improvements.
      `.trim()
    });
    
    return object;
  }
  
  /**
   * Calculate judge agreement
   */
  private calculateJudgeAgreement(evaluations: JudgeEvaluation[]): {
    overall: number;
    disagreements: string[];
  } {
    if (evaluations.length < 2) {
      return { overall: 1.0, disagreements: [] };
    }
    
    const overallScores = evaluations.map(e => e.overallScore);
    const agreement = this.calculateAgreement(overallScores);
    
    // Find dimensions with disagreement
    const dimensionGroups: Record<string, number[]> = {};
    evaluations.forEach(evaluation => {
      evaluation.dimensions.forEach(dim => {
        if (!dimensionGroups[dim.name]) {
          dimensionGroups[dim.name] = [];
        }
        dimensionGroups[dim.name].push(dim.score);
      });
    });
    
    const disagreements: string[] = [];
    Object.entries(dimensionGroups).forEach(([name, scores]) => {
      if (this.stdDev(scores) > 1.0) {
        disagreements.push(name);
      }
    });
    
    return { overall: agreement, disagreements };
  }
  
  /**
   * Calculate confidence
   */
  private calculateConfidence(
    judgeEvaluations: JudgeEvaluation[],
    centralScores: CentralRubricScores,
    judgeAgreement: number
  ): number {
    const avgJudgeConfidence = judgeEvaluations.length > 0
      ? this.mean(judgeEvaluations.map(e => e.confidence))
      : 0.5;
    
    const centralScoreValues = [
      centralScores.coherenceStructure,
      centralScores.fluency,
      centralScores.lexical,
      centralScores.technical,
      centralScores.softSkill,
      centralScores.cultural
    ];
    
    const centralConsistency = 1 - (this.stdDev(centralScoreValues) / 5);
    
    const confidence = (judgeAgreement * 0.5) + (avgJudgeConfidence * 0.3) + (centralConsistency * 0.2);
    
    return Math.round(confidence * 100) / 100;
  }
  
  /**
   * Calculate agreement metric
   */
  private calculateAgreement(scores: number[]): number {
    const stdDev = this.stdDev(scores);
    const agreement = Math.max(0, 1 - (stdDev / 2));
    return Math.round(agreement * 100) / 100;
  }
  
  // ========== Statistical Helper Methods ==========
  
  private mean(numbers: number[]): number {
    if (numbers.length === 0) return 0;
    const sum = numbers.reduce((acc, n) => acc + n, 0);
    return Math.round((sum / numbers.length) * 100) / 100;
  }
  
  private median(numbers: number[]): number {
    if (numbers.length === 0) return 0;
    const sorted = [...numbers].sort((a, b) => a - b);
    const mid = Math.floor(sorted.length / 2);
    if (sorted.length % 2 === 0) {
      return Math.round(((sorted[mid - 1] + sorted[mid]) / 2) * 100) / 100;
    }
    return sorted[mid];
  }
  
  private stdDev(numbers: number[]): number {
    if (numbers.length === 0) return 0;
    const avg = this.mean(numbers);
    const squareDiffs = numbers.map(value => Math.pow(value - avg, 2));
    const avgSquareDiff = this.mean(squareDiffs);
    return Math.round(Math.sqrt(avgSquareDiff) * 100) / 100;
  }

  }
}
```

```typescript
import { BatchEvaluationInput } from '@/types/evaluation';

abstract class BaseJudge {
  protected apiKey: string;
  
  constructor(apiKey: string) {
    this.apiKey = apiKey;
  }
  
  abstract evaluate(params: EvaluationParams): Promise<JudgeEvaluation>;
  
  /**
   * Evaluate multiple questions in batch
   */
  async evaluateMultiple(
    questions: BatchEvaluationInput[]
  ): Promise<JudgeEvaluation[]> {
    console.log(`${this.constructor.name}: Evaluating ${questions.length} questions...`);
    
    const start = Date.now();
    
    // Process all questions in parallel
    const results = await Promise.all(
      questions.map(q => this.evaluate({
        question: q.question,
        answer: q.answer,
        context: q.context
      }))
    );
    
    console.log(`${this.constructor.name}: Batch complete (${Date.now() - start}ms)`);
    return results;
  }
}
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import StarJudge from "@/lib/evaluation/judges/StarJudge";
import CompetencyJudge from "@/lib/evaluation/judges/CompetencyJudge";
import { Aggregator } from "@/lib/evaluation/Aggregator";
import { saveEvaluation } from "@/lib/actions/evaluation.action";
import { InterviewContext, BatchEvaluationInput } from "@/types/evaluation";

interface EvaluateInterviewRequest {
  interviewId: string;
  userId: string;
  fullTranscript: string;
  questions: string[];
  interviewContext: InterviewContext;
}

export async function POST(req: NextRequest) {
  const start = Date.now();

  try {
    const body: EvaluateInterviewRequest = await req.json();
    const {
      interviewId,
      userId,
      fullTranscript,
      questions,
      interviewContext
    } = body;

    // Validate required fields
    if (!interviewId || !fullTranscript || !questions || questions.length === 0) {
      return NextResponse.json(
        { error: "Missing required fields: interviewId, fullTranscript, questions" },
        { status: 400 }
      );
    }

    const apiKey = process.env.GOOGLE_GENERATIVE_AI_API_KEY;
    if (!apiKey) {
      console.error('GOOGLE_GENERATIVE_AI_API_KEY not found in environment');
      return NextResponse.json(
        { error: "API key not configured" },
        { status: 500 }
      );
    }

    console.log(`[evaluate-interview] Starting evaluation for ${interviewId}`);
    console.log(`[evaluate-interview] Questions: ${questions.length}, Transcript length: ${fullTranscript.length} chars`);

    // Prepare question objects for batch evaluation
    const questionInputs: BatchEvaluationInput[] = questions.map((q, idx) => ({
      questionId: `question-${idx}`,
      question: q,
      answer: "", // Will be extracted by Aggregator
      context: interviewContext
    }));

    // API CALL 1: StarJudge evaluates all questions
    console.log("[evaluate-interview] API Call 1/3: StarJudge...");
    const starJudge = new StarJudge(apiKey);
    const starEvaluations = await starJudge.evaluateMultiple(questionInputs);
    
    // API CALL 2: CompetencyJudge evaluates all questions
    console.log("[evaluate-interview] API Call 2/3: CompetencyJudge...");
    const competencyJudge = new CompetencyJudge(apiKey);
    const competencyEvaluations = await competencyJudge.evaluateMultiple(questionInputs);
    
    // Combine judge evaluations per question
    const judgeEvaluationsPerQuestion = questions.map((_, idx) => [
      starEvaluations[idx],
      competencyEvaluations[idx]
    ]);
    
    // API CALL 3: Aggregator processes everything
    console.log("[evaluate-interview] API Call 3/3: Aggregator...");
    const aggregator = new Aggregator();
    const aggregated = await aggregator.aggregateInterview(
      {
        fullTranscript,
        questions,
        interviewContext: interviewContext || {
          role: "Candidate",
          field: "General",
          stage: "experienced"
        },
        judgeEvaluationsPerQuestion
      },
      apiKey  // Pass API key to method, not constructor
    );

    // Build final evaluation document
    const evaluation = {
      interviewId,
      userId,
      questions: aggregated.questionEvaluations,
      centralRubricScores: aggregated.centralRubricScores,
      overallScore: aggregated.overallScore,
      strengths: aggregated.strengths,
      improvements: aggregated.improvements,
      judgeAgreement: aggregated.judgeAgreement,
      confidenceScore: aggregated.confidenceScore,
      createdAt: new Date().toISOString(),
      processingTimeMs: Date.now() - start
    };

    // Save to Firestore
    const evaluationId = await saveEvaluation(evaluation);

    console.log(`[evaluate-interview] Success! Evaluation ${evaluationId} (${Date.now() - start}ms)`);

    return NextResponse.json({
      success: true,
      evaluationId,
      preview: {
        overallScore: aggregated.overallScore,
        confidence: aggregated.confidenceScore,
        questionCount: questions.length
      }
    });
  } catch (error) {
    console.error("[evaluate-interview] Error:", error);
    return NextResponse.json(
      {
        error: error instanceof Error ? error.message : "Evaluation failed",
        details: error instanceof Error ? error.stack : undefined
      },
      { status: 500 }
    );
  }
}

export async function GET() {
  return NextResponse.json({
    status: "evaluate-interview endpoint ready",
    version: "2.0"
  });
}
```

```typescript
import { db } from "@/firebase/admin";
import { StoredEvaluation } from "@/types/evaluation";

const COLLECTION = "evaluations";

/**
 * Save evaluation to Firestore
 */
export async function saveEvaluation(
  evaluation: Omit<StoredEvaluation, "id">
): Promise<string> {
  const ref = await db.collection(COLLECTION).add(evaluation);
  return ref.id;
}

/**
 * Get evaluation by ID
 */
export async function getEvaluationById(
  id: string
): Promise<StoredEvaluation | null> {
  const doc = await db.collection(COLLECTION).doc(id).get();
  if (!doc.exists) return null;
  return { id: doc.id, ...doc.data() } as StoredEvaluation;
}

/**
 * Get all evaluations for an interview
 * Returns single evaluation (per-interview model)
 */
export async function getEvaluationByInterview(
  interviewId: string
): Promise<StoredEvaluation | null> {
  const snapshot = await db
    .collection(COLLECTION)
    .where("interviewId", "==", interviewId)
    .limit(1)
    .get();

  if (snapshot.empty) return null;

  const doc = snapshot.docs[0];
  return { id: doc.id, ...doc.data() } as StoredEvaluation;
}

/**
 * Get all evaluations for multiple interviews
 * (for dashboard, analytics, etc.)
 */
export async function getEvaluationsByInterview(
  interviewId: string
): Promise<StoredEvaluation[]> {
  const snapshot = await db
    .collection(COLLECTION)
    .where("interviewId", "==", interviewId)
    .orderBy("createdAt", "desc")
    .get();

  return snapshot.docs.map(
    (doc) => ({ id: doc.id, ...doc.data() } as StoredEvaluation)
  );
}
```
